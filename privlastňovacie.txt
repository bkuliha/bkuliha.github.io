        let currentGameQuestions = [];
        let currentIndex = 0;
        let score = 0;
        let userSelection = null; // Used for 'find' type
        let isAnswered = false;

        function startGame() {
            // Shuffle and pick 10
            currentGameQuestions = [...questionBank]
                .sort(() => 0.5 - Math.random())
                .slice(0, 10);
            
            currentIndex = 0;
            score = 0;
            isAnswered = false;

            document.getElementById('start-screen').classList.add('hidden');
            document.getElementById('end-screen').classList.add('hidden');
            document.getElementById('question-screen').classList.remove('hidden');
            document.getElementById('progress-container').classList.remove('hidden');
            
            updateScoreDisplay();
            showQuestion();
        }

        function showQuestion() {
            isAnswered = false;
            userSelection = null;
            
            const q = currentGameQuestions[currentIndex];
            const content = document.getElementById('question-content');
            const instruction = document.getElementById('instruction-text');
            const feedback = document.getElementById('feedback-area');
            const submitBtn = document.getElementById('submit-btn');
            const nextBtn = document.getElementById('next-btn');

            // Reset UI
            content.innerHTML = '';
            instruction.innerText = q.instruction;
            feedback.classList.add('hidden');
            submitBtn.classList.remove('hidden');
            nextBtn.classList.add('hidden');
            
            document.getElementById('question-count').innerText = `Otázka ${currentIndex + 1} z 10`;
            document.getElementById('progress-bar').style.width = `${(currentIndex / 10) * 100}%`;

            if (q.type === 'find') {
                const words = q.sentence.split(' ');
                words.forEach((word, idx) => {
                    const btn = document.createElement('button');
                    // Clean punctuation for display but keep it for visual flow
                    btn.innerText = word;
                    btn.className = "word-btn px-3 py-2 border-2 border-slate-200 rounded-lg text-xl font-medium m-1";
                    btn.onclick = () => selectWord(btn, word, idx);
                    content.appendChild(btn);
                });
            } 
            else if (q.type === 'choice') {
                const wrapper = document.createElement('div');
                wrapper.className = "flex flex-col items-center gap-4 w-full";
                
                const sentenceDisplay = document.createElement('div');
                sentenceDisplay.className = "text-2xl text-center mb-4 leading-relaxed";
                sentenceDisplay.innerHTML = q.sentence.replace('_____', '<span class="border-b-4 border-indigo-200 px-4">?</span>');
                wrapper.appendChild(sentenceDisplay);

                const btnGroup = document.createElement('div');
                btnGroup.className = "flex gap-4";
                q.options.forEach(opt => {
                    const btn = document.createElement('button');
                    btn.innerText = opt;
                    btn.className = "choice-btn px-8 py-3 bg-white border-2 border-indigo-200 rounded-xl text-xl font-bold hover:bg-indigo-50 transition";
                    btn.onclick = () => {
                        userSelection = opt;
                        // Visual selection
                        Array.from(btnGroup.children).forEach(b => b.classList.replace('bg-indigo-600', 'bg-white'));
                        Array.from(btnGroup.children).forEach(b => b.classList.replace('text-white', 'text-indigo-600'));
                        btn.classList.add('bg-indigo-600', 'text-white');
                    };
                    btnGroup.appendChild(btn);
                });
                wrapper.appendChild(btnGroup);
                content.appendChild(wrapper);
            }
            else if (q.type === 'fill') {
                const wrapper = document.createElement('div');
                wrapper.className = "flex flex-col items-center gap-4 w-full";
                
                const sentenceDisplay = document.createElement('div');
                sentenceDisplay.className = "text-2xl text-center mb-4 leading-relaxed";
                sentenceDisplay.innerHTML = q.sentence.replace('____', '<input type="text" id="fill-input" class="w-32 text-center border-b-4 border-indigo-500 outline-none bg-transparent font-bold text-indigo-700" placeholder="...">');
                wrapper.appendChild(sentenceDisplay);
                
                content.appendChild(wrapper);
                // Focus the input
                setTimeout(() => document.getElementById('fill-input').focus(), 100);
            }
        }

        function selectWord(btn, word, idx) {
            if (isAnswered) return;
            const allBtns = document.querySelectorAll('.word-btn');
            allBtns.forEach(b => b.classList.remove('selected', 'border-indigo-500'));
            btn.classList.add('selected', 'border-indigo-500');
            userSelection = word.replace(/[.,!?]/g, ""); // strip punctuation for check
        }

        function checkAnswer() {
            if (isAnswered) return;
            const q = currentGameQuestions[currentIndex];
            let isCorrect = false;
            let userVal = "";

            if (q.type === 'find') {
                if (!userSelection) return;
                isCorrect = userSelection.toLowerCase() === q.answer.toLowerCase();
                userVal = userSelection;
            } 
            else if (q.type === 'choice') {
                if (!userSelection) return;
                isCorrect = userSelection === q.answer;
                userVal = userSelection;
            }
            else if (q.type === 'fill') {
                const input = document.getElementById('fill-input');
                userVal = input.value.trim().toLowerCase();
                if (!userVal) return;
                isCorrect = userVal === q.answer.toLowerCase();
            }

            isAnswered = true;
            const feedbackArea = document.getElementById('feedback-area');
            const feedbackText = document.getElementById('feedback-text');
            const correctDisplay = document.getElementById('correct-answer-display');
            const submitBtn = document.getElementById('submit-btn');
            const nextBtn = document.getElementById('next-btn');

            if (isCorrect) {
                score++;
                feedbackArea.className = "rounded-2xl p-4 mb-6 text-center font-bold bg-emerald-100 text-emerald-800 correct-anim";
                feedbackText.innerText = "Výborne! To je správne. ✨";
                correctDisplay.innerText = "";
            } else {
                feedbackArea.className = "rounded-2xl p-4 mb-6 text-center font-bold bg-rose-100 text-rose-800";
                feedbackText.innerText = "To nie je celkom ono. ✍️";
                correctDisplay.innerText = `Správna odpoveď: ${q.answer}`;
            }

            feedbackArea.classList.remove('hidden');
            submitBtn.classList.add('hidden');
            nextBtn.classList.remove('hidden');
            updateScoreDisplay();
        }

        function updateScoreDisplay() {
            document.getElementById('current-score').innerText = `Skóre: ${score}`;
        }

        function nextQuestion() {
            currentIndex++;
            if (currentIndex < 10) {
                showQuestion();
            } else {
                showEndScreen();
            }
        }

        function showEndScreen() {
            document.getElementById('question-screen').classList.add('hidden');
            document.getElementById('progress-container').classList.add('hidden');
            const endScreen = document.getElementById('end-screen');
            endScreen.classList.remove('hidden');
            
            const finalScore = document.getElementById('final-score');
            finalScore.innerText = `${score} / 10`;
            
            const finalMsg = document.getElementById('final-message');
            const emoji = document.getElementById('final-emoji');
            
            if (score === 10) {
                emoji.innerText = "👑";
                finalMsg.innerText = "Dokonalé! Si absolútny majster zámen.";
            } else if (score >= 8) {
                emoji.innerText = "🌟";
                finalMsg.innerText = "Skvelý výkon! Gramatiku máš v malíčku.";
            } else if (score >= 5) {
                emoji.innerText = "👍";
                finalMsg.innerText = "Dobrý pokus, ale ešte máme čo trénovať.";
            } else {
                emoji.innerText = "📚";
                finalMsg.innerText = "Nevzdávaj to. Skús si prejsť teóriu a zahraj si znova.";
            }
        }
